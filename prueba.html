# Obtener pares únicos
muertes_pares = muertes[['DPNOM', 'DPMP']].drop_duplicates()
poblacion_pares = poblacion[['DPNOM', 'DPMP']].drop_duplicates()

# Comparar: pares en muertes pero no en población
solo_muertes = muertes_pares.merge(poblacion_pares, on=['DPNOM', 'DPMP'], how='left', indicator=True)
solo_muertes = solo_muertes[solo_muertes['_merge'] == 'left_only']

# Comparar: pares en población pero no en muertes
solo_poblacion = poblacion_pares.merge(muertes_pares, on=['DPNOM', 'DPMP'], how='left', indicator=True)
solo_poblacion = solo_poblacion[solo_poblacion['_merge'] == 'left_only']

muertes_set = set(tuple(x) for x in muertes[['DPNOM', 'DPMP']].drop_duplicates().values)
poblacion_set = set(tuple(x) for x in poblacion[['DPNOM', 'DPMP']].drop_duplicates().values)

# Diferencias
solo_en_muertes = muertes_set - poblacion_set
solo_en_poblacion = poblacion_set - muertes_set
